import { setStatusBarBackgroundColor } from 'expo-status-bar';
import React, {useState} from 'react';
import { TextInput, StyleSheet, Text, View, KeyboardAvoidingView, Platform, TouchableOpacity, SafeAreaView, Keyboard } from 'react-native';

const Bookmark = () => {

    const [bookmark, setbookmark] = useState();
    const [bookmarkItems, setBookmarkItems] = useState([]);

    const handleBookmark = () => {
        Keyboard.dismiss();
        setBookmarkItems([...bookmarkItems, bookmark]);
        // setbookmark(null);
        
        // Add the following to the bookmark page <View>
        // bookmarkItems.map((Item) => { return <Task key={index} text={item} /> })
    }

    return (
         <KeyboardAvoidingView
            behavior={Platform.OS === "ios" ? "padding" : "height" }
            style={styles.writeBookMarkWrapper}>
            <TextInput style={styles.input} placeholder={"Leave a bookmark!"} value={bookmark} onChangeText={text => setbookmark(text)}/>
            
            <TouchableOpacity onPress={() => handleBookmark()}>
                <View style={styles.addWrapper}>
                     <Text style={styles.addText}>+</Text>
                </View>
            </TouchableOpacity>
        </KeyboardAvoidingView>
    )
}

const styles = StyleSheet.create({
    writeBookMarkWrapper: {
        position: 'absolute',
        bottom: 30,
        width: '100%',
        flexDirection: 'row',
        justifyContent: 'space-around',
        alignItems: 'center',
    },
    input: {

        width: 60,
